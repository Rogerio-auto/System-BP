## CodeBlock — TypeScript

<CodeBlock lang="ts" title="Buscar leads por status">
{`import { api } from '@/lib/api';

export async function fetchLeads(status: 'new' | 'qualifying' | 'closed_won') {
  return api.get<LeadResponse[]>('/api/leads', {
    params: { status, limit: 25 },
  });
}`}
